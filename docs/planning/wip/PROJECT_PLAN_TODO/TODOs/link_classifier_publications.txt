import httpx, mimetypes

RANGE = "bytes=0-65535"          # first 64 kB

async def head_and_chunk(url: str) -> tuple[str, bytes]:
    async with httpx.AsyncClient(timeout=8.0, follow_redirects=True) as c:
        r = await c.get(url, headers={"Range": RANGE})
    ctype = r.headers.get("content-type") or mimetypes.guess_type(url)[0] or ""
    return ctype.lower(), r.content           # 206 Partial is fine :contentReference[oaicite:2]{index=2}
